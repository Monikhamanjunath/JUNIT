package searchApp;

public class LinearSearch {

    /**
     * Searches for a target in an array.
     * @return the index of the target, or -1 if not found.
     */
    public int find(int[] arr, int target) {
        if (arr == null) return -1;
        
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                return i; // Found it!
            }
        }
        return -1; // Not found
    }

    public static void main(String[] args) {
        LinearSearch ls = new LinearSearch();
        int[] numbers = {10, 20, 30, 40, 50};
        System.out.println("Searching for 30: Index " + ls.find(numbers, 30));
    }
}

package searchApp;

import static org.junit.Assert.*;
import org.junit.Before;
import org.junit.Test;

public class LinearSearchTest {

    private LinearSearch ls;

    @Before
    public void setUp() {
        // Initialize the logic object
        ls = new LinearSearch();
    }

    @Test
    public void testElementExists() {
        int[] data = {5, 8, 12, 20};
        // Use ls.find to avoid "undefined method" error
        assertEquals(2, ls.find(data, 12));
    }

    @Test
    public void testElementMissing() {
        int[] data = {1, 2, 3};
        assertEquals(-1, ls.find(data, 99));
    }

    @Test
    public void testEmptyArray() {
        int[] data = {};
        assertEquals(-1, ls.find(data, 10));
    }

    @Test
    public void testNullArray() {
        assertEquals(-1, ls.find(null, 5));
    }
}
