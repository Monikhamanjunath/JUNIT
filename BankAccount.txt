BankService.java
package BankService;





public class BankService {

    // Nested class (can be public or private, doesn't matter for functionality)
    static class BankAccount {
        private double balance;

        public BankAccount(double initialBalance) {
            if (initialBalance < 0) {
                throw new IllegalArgumentException("Initial balance cannot be negative");
            }
            this.balance = initialBalance;
        }

        public void deposit(double amount) {
            if (amount <= 0) {
                throw new IllegalArgumentException("Deposit must be positive");
            }
            balance += amount;
        }

        public void withdraw(double amount) {
            if (amount <= 0 || amount > balance) {
                throw new IllegalArgumentException("Invalid withdrawal");
            }
            balance -= amount;
        }

        public double getBalance() {
            return balance;
        }
    }

    // The main entry point
    public static void main(String[] args) {
        // Create an instance of the BankAccount class
        BankAccount acc = new BankAccount(5000);
        
        System.out.println("Initial Balance: " + acc.getBalance());

        // Perform operations
        acc.deposit(700);
        System.out.println("Balance after: " + acc.getBalance());

        acc.withdraw(100);
        System.out.println("Balance after withdraw of 300: " + acc.getBalance());

        // Print final result
        System.out.println("Final Balance: " + acc.getBalance());
    }
}

BankServiceTest.java

package BankService;

import static org.junit.Assert.*;
import org.junit.Before;
import org.junit.Test;

public class BankServiceTest {

    private BankService.BankAccount account;

    @Before
    public void setUp() {
        // Initialize with 5000 before each test
        account = new BankService.BankAccount(5000);
    }

    @Test
    public void testDeposit() {
        account.deposit(500);
        // We expect 5000 + 500 = 5500
        assertEquals(5500.0, account.getBalance(), 0.001);
    }

    @Test
    public void testWithdraw() {
        account.withdraw(1000);
        // We expect 5000 - 1000 = 4000
        assertEquals(4000.0, account.getBalance(), 0.001);
    }

    @Test(expected = IllegalArgumentException.class)
    public void testInvalidWithdrawal() {
        // This should throw an exception because it's more than the balance
        account.withdraw(6000);
    }

    @Test(expected = IllegalArgumentException.class)
    public void testNegativeDeposit() {
        // This should throw an exception
        account.deposit(-100);
    }

    @Test
    public void testInitialBalance() {
        assertEquals(5000.0, account.getBalance(), 0.001);
    }
}
